37  virtual exrBool Scatter(const Ray& incomingRay, const Interaction& hitInfo, exrVector3& attenuation, Ray& scattered, exrFloat& pdf) const override
38  {
39  exrVector3 reflectedRay = Reflect(incomingRay.m_Direction, hitInfo.m_Normal);
40 
41  exrFloat ni_over_nt;
42  attenuation = m_Albedo;
43 
44  exrVector3 outNormal;
45  exrVector3 refractedRay;
46 
47  // for Fresnel calculations
48  exrFloat cosine;
49  exrFloat reflectionProbability;
50 
51  // entering material
52  if (Dot(incomingRay.m_Direction, hitInfo.m_Normal) > 0.0f)
53  {
54  outNormal = -hitInfo.m_Normal;
55  ni_over_nt = m_RefractiveIndex;
56  cosine = m_RefractiveIndex * Dot(incomingRay.m_Direction, hitInfo.m_Normal) / incomingRay.m_Direction.Magnitude();
57  }
58  else // exiting material
59  {
60  outNormal = hitInfo.m_Normal;
61  ni_over_nt = 1.0f / m_RefractiveIndex;
62  cosine = -Dot(incomingRay.m_Direction, hitInfo.m_Normal) / incomingRay.m_Direction.Magnitude();
63  }
64 
65  if (Refract(incomingRay.m_Direction, outNormal, ni_over_nt, refractedRay))
66  {
67  // refraction
68  reflectionProbability = SchlickFresnelApproximation(cosine, m_RefractiveIndex);
69  }
70  else
71  {
72  // total internal reflection
73  reflectionProbability = 1.0f;
74  }
75 
76  // Reflection based on Fresnel approximation
77  if (Random::Random01() < reflectionProbability)
78  {
79  scattered = Ray(hitInfo.m_Point, reflectedRay);
80  }
81  else
82  {
83  scattered = Ray(hitInfo.m_Point, refractedRay);
84  }
85 
86  return true;
87  };
