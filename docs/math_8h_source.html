64 inline exrBool exrQuadratic(exrFloat a, exrFloat b, exrFloat c, exrFloat* t0, exrFloat* t1)
65 {
66  exrFloat discriminator = b * b - 4 * a * c;
67 
68  if (discriminator < 0)
69  return false;
70 
71  exrFloat droot = sqrt(discriminator);
72 
73  // Compute quadratic values
74  *t0 = -0.5f * a * (b - droot);
75  *t1 = -0.5f * a * (b + droot);
76 
77  if (*t0 > * t1) std::swap(*t0, *t1);
78  return true;
79 }
