44  Camera(exrPoint3 position, exrPoint3 lookat, exrVector3 up, exrFloat vfov, exrFloat aspect, exrFloat aperture, exrFloat focusDist) {
45  // virtual lens to simulate defocus blur
46  lensRadius = aperture / 2.0f;
47 
48  exrFloat theta = vfov * exrFloat(EXR_M_PI) / 180.f;
49  exrFloat halfHeight = tan(theta / 2.0f);
50  exrFloat halfWidth = aspect * halfHeight;
51 
52  m_Position = position;
53  w = (lookat - position).Normalized(); // look in positive -z axis, RH coordinate system
54  u = (Cross(w, up)).Normalized();
55  v = Cross(u, w);
56 
57  m_Min = position - halfWidth * focusDist * u - halfHeight * focusDist * v + w * focusDist;
58  m_HorizontalStep = 2.0f * halfWidth * focusDist* u;
59  m_VerticalStep = 2.0f * halfHeight * focusDist * v;
60  }

66  Ray GetViewRay(exrFloat s, exrFloat t)
67  {
68  exrVector3 rd = lensRadius * Random::RandomInUnitDisc();
69  exrVector3 offset = u * rd.x + v * rd.y;
70  return Ray(m_Position + offset, m_Min + s * m_HorizontalStep + t * m_VerticalStep - m_Position - offset);
71  }
