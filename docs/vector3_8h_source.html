149 template<class T>
150 inline bool Refract(const Vector3<T>& v, const Vector3<T>& n, float ni_over_nt, Vector3<T>& refracted)
151 {
152  Vector3<T> vNorm = v.Normalized();
153  float dt = Dot(vNorm, n);
154  float discriminant = 1.0f - ni_over_nt * ni_over_nt * (1.0f - dt * dt);
155 
156  if (discriminant > 0)
157  {
158  refracted = ni_over_nt * (vNorm - n * dt) - n * sqrt(discriminant);
159  return true;
160  }
161 
162  return false;
163 }
