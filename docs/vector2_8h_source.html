127 template<class T>
128 inline bool Refract(const Vector2<T>& v, const Vector2<T>& n, float ni_over_nt, Vector2<T>& refracted)
129 {
130  Vector2<T> vNorm = v.Normalized();
131  float dt = Dot(vNorm, n);
132  float discriminant = 1.0f - ni_over_nt * ni_over_nt * (1.0f - dt * dt);
133 
134  if (discriminant > 0)
135  {
136  refracted = ni_over_nt * (vNorm - n * dt) - n * sqrt(discriminant);
137  return true;
138  }
139 
140  return false;
141 }
